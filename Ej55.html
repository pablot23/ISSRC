El codigo es el siguiente:

 
            let color = "jhjhjk";
            if (color === "Rojo") {
            console.log("El color es Rojo");
            } 
            else if (color === "Verde") {
            console.log("El color es Verde");
            } 
            else if (color === "Azul") {
            console.log("El color es Azul");
            } 
            else {
            console.log("El color no es Rojo, Verde ni Azul");
