Crea una variable llamada `diaSemana` y asígnale el nombre de un día de la semana. Utiliza una declaración `switch` para mostrar un mensaje que indique si es un "Día laboral" o "Fin de semana"

El codigo es el siguiente:

 
    let diasemana = prompt ("ingresar un dia de la semana");
    switch (diasemana) {
    case "lunes":
        console.log("día laboral");
        break;
    case "martes":
        console.log("día laboral");
        break;
    case "miercoles":
        console.log("día laboral");
        break;
    case "jueves":
        console.log("día laboral");
        break;
    case "viernes":
        console.log("el cuerpo lo sabe");
        break;   
    case "sábado":
        console.log("fin de");
        break;
    case "domingo":
        console.log("fin de");
        break;                                                         
    }
